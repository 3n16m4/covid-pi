sudo echo dtparam=i2c_baudrate=1000000 > /boot/config.txt

Installation:

git clone git@github.com:3n16m4/covid-pi.git && cd covid-pi
mkdir build && cd build
cmake .. -DCMAKE_TOOLCHAIN_FILE=../cmake/Toolchain-rpi.cmake && make -j
./covid-pi
